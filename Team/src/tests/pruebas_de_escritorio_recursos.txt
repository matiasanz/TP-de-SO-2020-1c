#include "team.h"
#include "hilos del team/hilos_team.h"
#include <commons/string.h>
//#include "tests/tests_team.h"

int main(void) {
	char*string = "1|22|333|4444|22|4444|333|4444|333|4444";
	matriz_recursos matriz = recursos_from_string(string);
	recursos_mostrar(matriz);

	puts("Se agrega R1");
	recursos_agregar_recurso(matriz, "1");
	recursos_mostrar(matriz);


	puts("Le resto todos menos ese");
	matriz_recursos otraMatriz = recursos_from_string(string);
	recursos_restar_recursos_a(matriz, otraMatriz);
	recursos_mostrar(matriz);

	puts("La otra sigue igual");

	recursos_mostrar(otraMatriz);

	puts("Se forzara un error");
	recursos_agregar_validado(matriz, NULL);

	recursos_destroy(matriz);

	return EXIT_SUCCESS;
}
